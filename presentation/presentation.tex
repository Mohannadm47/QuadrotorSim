\section{Outline and Introduction}

\frame{
  \frametitle{Introduction}
  
\textbf{Goal}: Build a stability and tracking controller for a quadcopter.
\begin{itemize}
\item Stability: maintain orientation and hover at a particular altitude.
\item Tracking: follow trajectories given.
\end{itemize}
\textbf{Subgoal}: Build a simulator for the quadrotor in matlab.
\textbf{Result}: Built an inverse dynamics controller that can
stabilize the quadcopter and track paths.
}
